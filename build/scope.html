public render() {
  const {
    points,
    width,
    height,
    xUnit,
    yUnit,
    colors = DEFAULT_COLORS
  } = this.props

  const xVals = points.map(d => d.x)
  const xScale = getScale(xVals, xUnit)

  const yVals = points.map(d => d.y)
  const yScale = getScale(yVals, yUnit)

  const scaledPoints = points.map(p => {
    return {
      x: xScale(p.x)
      y: yScale(p.y)
    }
  })
}
